        // NUEVA FUNCIÓN: Cargar festivos desde Firebase
        async function loadHolidays() {
            try {
                const snapshot = await db.collection('holidays').orderBy('date').get();
                
                state.holidays = [];
                snapshot.forEach(doc => {
                    const holiday = { 
                        id: doc.id, 
                        ...doc.data(),
                        date: convertTimestampToDate(doc.data().date)
                    };
                    state.holidays.push(holiday);
                });
                
                updateHolidayStats();
                // Si estamos en la pestaña de admin, renderizar calendario
                if (elements.adminTab.classList.contains('active') && state.isAdmin) {
                    renderCalendar();
                }
                return true;
                
            } catch (error) {
                console.error("Error cargando festivos:", error);
                return false;
            }
        }

        // NUEVA FUNCIÓN: Actualizar estadísticas de festivos
        function updateHolidayStats() {
            const today = new Date();
            const currentYear = today.getFullYear();
            const currentMonth = today.getMonth() + 1;
            
            const holidaysThisMonth = state.holidays.filter(holiday => {
                const holidayDate = new Date(holiday.date);
                return holidayDate.getFullYear() === currentYear && 
                       (holidayDate.getMonth() + 1) === currentMonth;
            }).length;
            
            const holidaysThisYear = state.holidays.filter(holiday => {
                const holidayDate = new Date(holiday.date);
                return holidayDate.getFullYear() === currentYear;
            }).length;
            
            elements.festivosMes.textContent = holidaysThisMonth.toString();
            elements.festivosAno.textContent = holidaysThisYear.toString();
        }

        // NUEVA FUNCIÓN: Renderizar calendario
        function renderCalendar() {
            const calendarDate = state.currentCalendarDate;
            const year = calendarDate.getFullYear();
            const month = calendarDate.getMonth();
            
            // Obtener primer día del mes
            const firstDay = new Date(year, month, 1);
            // Obtener último día del mes
            const lastDay = new Date(year, month + 1, 0);
            // Días en el mes
            const daysInMonth = lastDay.getDate();
            // Día de la semana del primer día (0 = Domingo, 1 = Lunes, ...)
            const firstDayOfWeek = firstDay.getDay();
            
            // Nombres de meses y días
            const monthNames = ["Enero", "Febrero", "Marzo", "Abril", "Mayo", "Junio",
                               "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre"];
            
            const dayNames = ["Dom", "Lun", "Mar", "Mié", "Jue", "Vie", "Sáb"];
            
            // Crear HTML del calendario
            let calendarHTML = `
                <div class="calendar-header">
                    <div class="calendar-nav">
                        <button id="prev-month-btn"><i class="fas fa-chevron-left"></i></button>
                    </div>
                    <div class="calendar-month-year">
                        ${monthNames[month]} ${year}
                    </div>
                    <div class="calendar-nav">
                        <button id="next-month-btn"><i class="fas fa-chevron-right"></i></button>
                    </div>
                </div>
                <div class="calendar-weekdays">
            `;
            
            // Añadir encabezados de días
            dayNames.forEach(day => {
                calendarHTML += `<div class="calendar-weekday">${day}</div>`;
            });
            
            calendarHTML += `</div><div class="calendar-days">`;
            
            // Añadir días vacíos antes del primer día del mes
            for (let i = 0; i < firstDayOfWeek; i++) {
                const prevMonthDay = new Date(year, month, -i);
                calendarHTML += createDayElement(prevMonthDay, true);
            }
            
            // Añadir días del mes actual
            const today = new Date();
            for (let day = 1; day <= daysInMonth; day++) {
                const currentDate = new Date(year, month, day);
                const isToday = currentDate.getDate() === today.getDate() &&
                               currentDate.getMonth() === today.getMonth() &&
                               currentDate.getFullYear() === today.getFullYear();
                
                calendarHTML += createDayElement(currentDate, false, isToday);
            }
            
            // Añadir días vacíos después del último día del mes
            const totalCells = 42; // 6 semanas * 7 días
            const daysSoFar = firstDayOfWeek + daysInMonth;
            for (let i = 1; i <= (totalCells - daysSoFar); i++) {
                const nextMonthDay = new Date(year, month + 1, i);
                calendarHTML += createDayElement(nextMonthDay, true);
            }
            
            calendarHTML += `</div>`;
            
            elements.holidayCalendar.innerHTML = calendarHTML;
            
            // Añadir event listeners a los botones de navegación
            document.getElementById('prev-month-btn').addEventListener('click', () => {
                state.currentCalendarDate.setMonth(state.currentCalendarDate.getMonth() - 1);
                renderCalendar();
            });
            
            document.getElementById('next-month-btn').addEventListener('click', () => {
                state.currentCalendarDate.setMonth(state.currentCalendarDate.getMonth() + 1);
                renderCalendar();
            });
        }

        // NUEVA FUNCIÓN: Crear elemento de día para el calendario
        function createDayElement(date, isOtherMonth, isToday = false) {
            const dayNumber = date.getDate();
            const dayOfWeek = date.getDay();
            const isWeekend = dayOfWeek === 0 || dayOfWeek === 6;
            
            // Verificar si es festivo
            const holiday = state.holidays.find(h => {
                const holidayDate = new Date(h.date);
                return holidayDate.getDate() === date.getDate() &&
                       holidayDate.getMonth() === date.getMonth() &&
                       holidayDate.getFullYear() === date.getFullYear();
            });
            
            let dayClasses = 'calendar-day';
            if (isOtherMonth) dayClasses += ' day-other-month';
            if (isToday) dayClasses += ' day-today';
            if (isWeekend) dayClasses += ' day-weekend';
            if (holiday) dayClasses += ' day-holiday';
            
            let holidayHTML = '';
            if (holiday) {
                holidayHTML = `
                    <div class="holiday-indicator">Festivo</div>
                    <div class="holiday-name">${holiday.name}</div>
                `;
            }
            
            return `
                <div class="${dayClasses}" data-date="${formatDateForInput(date)}" data-full-date="${date.toISOString()}">
                    <div class="day-number">${dayNumber}</div>
                    ${holidayHTML}
                </div>
            `;
        }

        // NUEVA FUNCIÓN: Abrir modal para añadir/editar festivo
        function openHolidayModal(date = null, holiday = null) {
            const today = new Date();
            const todayStr = formatDateForInput(today);
            
            if (holiday) {
                // Modo edición
                state.editingHoliday = holiday;
                elements.holidayModalTitle.innerHTML = '<i class="fas fa-edit"></i> Editar Día Festivo';
                elements.holidayDateInput.value = formatDateForInput(new Date(holiday.date));

                elements.holidayType.value = holiday.type || 'nacional';
                elements.holidayRecurring.checked = holiday.recurring === true;
                elements.deleteHolidayBtn.style.display = 'block';

                state.editingHoliday = null;
                elements.holidayModalTitle.innerHTML = '<i class="fas fa-calendar-plus"></i> Nuevo Día Festivo';
                elements.holidayDateInput.value = date || todayStr;
                elements.holidayNameInput.value = '';
                elements.holidayType.value = 'nacional';
                elements.holidayRecurring.checked = false;
                elements.deleteHolidayBtn.style.display = 'none';
            }
            
            elements.holidayModal.classList.add('active');
        }

        // NUEVA FUNCIÓN: Guardar festivo
        async function saveHoliday() {
            const dateStr = elements.holidayDateInput.value;
            const name = elements.holidayNameInput.value.trim();
            const type = elements.holidayType.value;
            const recurring = elements.holidayRecurring.checked;
            
            if (!dateStr || !name) {
                showToast("Complete la fecha y el nombre del festivo", "warning");
                return;
            }
            
            try {
                const holidayDate = new Date(dateStr);
                const holidayId = state.editingHoliday ? state.editingHoliday.id : `HOLIDAY_${Date.now()}`;
                
                const holidayData = {
                    id: holidayId,
                    date: holidayDate,
                    name: name,
                    type: type,
                    recurring: recurring,
                    year: holidayDate.getFullYear(),
                    month: holidayDate.getMonth() + 1,
                    day: holidayDate.getDate(),
                    updatedAt: new Date()
                };
                
                if (state.editingHoliday) {
                    // Actualizar festivo existente
                    await db.collection('holidays').doc(holidayId).update(holidayData);
                    showToast("Festivo actualizado correctamente", "success");
                } else {
                    // Crear nuevo festivo
                    holidayData.createdAt = new Date();
                    await db.collection('holidays').doc(holidayId).set(holidayData);
                    showToast("Festivo creado correctamente", "success");
                }
                
                closeHolidayModal();
                await loadHolidays();
                if (elements.adminTab.classList.contains('active') && state.isAdmin) {
                    renderCalendar();
                }
                
            } catch (error) {
                console.error("Error guardando festivo:", error);
                showToast(`Error: ${error.message}`, "error");
            }
        }

        // NUEVA FUNCIÓN: Eliminar festivo
        async function deleteHoliday() {
            if (!state.editingHoliday) return;
            
            const confirmed = await showConfirmationModal(
                'Eliminar Festivo',
                `¿Está seguro de eliminar el festivo "${state.editingHoliday.name}" del ${formatDate(new Date(state.editingHoliday.date))}?`,
                'fas fa-trash',
                'danger'
            );
            
            if (!confirmed) {
                showToast("Eliminación cancelada", "info");
                return;
            }
            
            try {
                await db.collection('holidays').doc(state.editingHoliday.id).delete();
                showToast("Festivo eliminado correctamente", "success");
                
                closeHolidayModal();
                await loadHolidays();
                if (elements.adminTab.classList.contains('active') && state.isAdmin) {
                    renderCalendar();
                }
                
            } catch (error) {
                console.error("Error eliminando festivo:", error);
                showToast(`Error: ${error.message}`, "error");
            }
        }

        // NUEVA FUNCIÓN: Mostrar todos los festivos
        function showAllHolidays() {
            elements.holidayListContainer.style.display = 'block';
            elements.holidayList.innerHTML = '';
            
            if (state.holidays.length === 0) {
                elements.holidayList.innerHTML = `
                    <div class="empty-state" style="padding: 20px;">
                        <i class="fas fa-calendar-times"></i>
                        <p>No hay días festivos registrados</p>
                    </div>
                `;
                return;
            }
            
            // Ordenar festivos por fecha
            const sortedHolidays = [...state.holidays].sort((a, b) => 
                new Date(a.date) - new Date(b.date)
            );
            
            sortedHolidays.forEach(holiday => {
                const holidayDate = new Date(holiday.date);
                const holidayItem = document.createElement('div');
                holidayItem.className = 'holiday-list-item';
                holidayItem.innerHTML = `
                    <div class="holiday-date">${formatDate(holidayDate)}</div>
                    <div class="holiday-list-name">${holiday.name}</div>
                    <div style="color: #7f8c8d; font-size: 0.9rem;">${holiday.type || 'nacional'}</div>
                    <div class="employee-actions">
                        <button class="btn btn-small btn-warning edit-holiday-btn" data-id="${holiday.id}">
                            <i class="fas fa-edit"></i>
                        </button>
                        <button class="btn btn-small btn-danger delete-holiday-list-btn" data-id="${holiday.id}">
                            <i class="fas fa-trash"></i>
                        </button>
                    </div>
                `;
                
                elements.holidayList.appendChild(holidayItem);
            });
            
            // Añadir event listeners a los botones
            document.querySelectorAll('.edit-holiday-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    e.stopPropagation();
                    const holidayId = btn.getAttribute('data-id');
                    const holiday = state.holidays.find(h => h.id === holidayId);
                    if (holiday) {
                        openHolidayModal(null, holiday);
                    }
                });
            });
            
            document.querySelectorAll('.delete-holiday-list-btn').forEach(btn => {
                btn.addEventListener('click', async (e) => {
                    e.stopPropagation();
                    const holidayId = btn.getAttribute('data-id');
                    const holiday = state.holidays.find(h => h.id === holidayId);
                    
                    if (holiday && confirm(`¿Eliminar el festivo "${holiday.name}"?`)) {
                        try {
                            await db.collection('holidays').doc(holidayId).delete();
                            showToast("Festivo eliminado", "success");
                            await loadHolidays();
                            if (elements.adminTab.classList.contains('active') && state.isAdmin) {
                                renderCalendar();
                            }
                            showAllHolidays(); // Refrescar lista
                        } catch (error) {
                            console.error("Error eliminando festivo:", error);
                            showToast(`Error: ${error.message}`, "error");
                        }
                    }
                });
            });
            
            // Abrir modal si no está abierto
            if (!elements.holidayModal.classList.contains('active')) {
                openHolidayModal();
            }
        }

        // NUEVA FUNCIÓN: Añadir festivo en lote
        async function addBulkHoliday() {
            const month = elements.bulkMonth.value;
            const year = elements.bulkYear.value;
            const day = elements.bulkDay.value;
            const name = elements.bulkName.value.trim();
            
            if (!month || !year || !day || !name) {
                showToast("Complete todos los campos para añadir festivo", "warning");
                return;
            }
            
            const dayNum = parseInt(day);
            if (dayNum < 1 || dayNum > 31) {
                showToast("El día debe estar entre 1 y 31", "warning");
                return;
            }
            
            try {
                // Verificar si la fecha es válida
                const date = new Date(year, month - 1, day);
                if (date.getMonth() + 1 != month || date.getDate() != dayNum) {
                    showToast("Fecha inválida para el mes seleccionado", "error");
                    return;
                }
                
                // Verificar si ya existe un festivo en esa fecha
                const existingHoliday = state.holidays.find(h => {
                    const holidayDate = new Date(h.date);
                    return holidayDate.getFullYear() == year &&
                           holidayDate.getMonth() + 1 == month &&
                           holidayDate.getDate() == dayNum;
                });
                
                if (existingHoliday) {
                    showToast(`Ya existe un festivo (${existingHoliday.name}) en esa fecha`, "warning");
                    return;
                }
                
                const holidayId = `HOLIDAY_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
                
                await db.collection('holidays').doc(holidayId).set({
                    id: holidayId,
                    date: date,
                    name: name,
                    type: 'nacional',
                    recurring: false,
                    year: parseInt(year),
                    month: parseInt(month),
                    day: dayNum,
                    createdAt: new Date(),
                    updatedAt: new Date()
                });
                
                showToast(`Festivo "${name}" añadido para el ${day}/${month}/${year}`, "success");
                
                // Limpiar campos
                elements.bulkDay.value = '';
                elements.bulkName.value = '';
                
                // Actualizar
                await loadHolidays();
                if (elements.adminTab.classList.contains('active') && state.isAdmin) {
                    renderCalendar();
                }
                
            } catch (error) {
                console.error("Error añadiendo festivo en lote:", error);
                showToast(`Error: ${error.message}`, "error");
            }
        }
